    <script>
      // -------------------------
      // DOM
      // -------------------------
      const projectsList = document.getElementById("projectsList");
      const chatsList = document.getElementById("chatsList");
      const chatInner = document.getElementById("chatInner");
      const chatScroll = document.getElementById("chatScroll");
      const input = document.getElementById("input");
      const sendBtn = document.getElementById("sendBtn");
      const newProjectBtn = document.getElementById("newProjectBtn");
      const newChatBtn = document.getElementById("newChatBtn");
      const renameChatBtn = document.getElementById("renameChatBtn");
      const topTitle = document.getElementById("topTitle");
      const statusEl = document.getElementById("status");
      const projectCount = document.getElementById("projectCount");
      const chatCount = document.getElementById("chatCount");
      const searchChats = document.getElementById("searchChats");

      // Modal DOM
      const modalBackdrop = document.getElementById("modalBackdrop");
      const modalTitle = document.getElementById("modalTitle");
      const modalLabel = document.getElementById("modalLabel");
      const modalInput = document.getElementById("modalInput");
      const modalError = document.getElementById("modalError");
      const modalOk = document.getElementById("modalOk");
      const modalCancel = document.getElementById("modalCancel");
      const modalClose = document.getElementById("modalClose");

      const editProjectContextBtn = document.getElementById(
        "editProjectContextBtn"
      );

      // -------------------------
      // State
      // -------------------------
      let projects = [];
      let chats = [];
      let selectedProjectId = null;
      let selectedChatId = null;
      let modalOnConfirm = null;

      function setStatus(t) {
        statusEl.textContent = t;
      }

      // -------------------------
      // Utils
      // -------------------------
      function escapeHtml(s) {
        return (s ?? "")
          .replaceAll("&", "&amp;")
          .replaceAll("<", "&lt;")
          .replaceAll(">", "&gt;");
      }

      function autoResize() {
        input.style.height = "auto";
        input.style.height = Math.min(input.scrollHeight, 180) + "px";
      }

      // -------------------------
      // Modal
      // -------------------------
      function openModal({
        title,
        label,
        placeholder,
        okText,
        defaultValue,
        onConfirm,
      }) {
        modalTitle.textContent = title || "Acción";
        modalLabel.textContent = label || "Nombre";
        modalInput.placeholder = placeholder || "";
        modalInput.value = defaultValue || "";
        modalOk.textContent = okText || "Aceptar";

        modalError.textContent = "";
        modalError.classList.add("hidden");

        modalOnConfirm = onConfirm;
        modalBackdrop.classList.remove("hidden");
        setTimeout(() => modalInput.focus(), 0);
      }

      function closeModal() {
        modalBackdrop.classList.add("hidden");
        modalOnConfirm = null;
      }

      function showModalError(msg) {
        modalError.textContent = msg;
        modalError.classList.remove("hidden");
      }

      async function loadProject(projectId) {
        const data = await apiGet(`/api/projects/${projectId}`);
        return data.project;
      }

      modalCancel.onclick = closeModal;
      modalClose.onclick = closeModal;
      modalBackdrop.addEventListener("click", (e) => {
        if (e.target === modalBackdrop) closeModal();
      });
      modalInput.addEventListener("keydown", (e) => {
        if (e.key === "Enter") modalOk.click();
        if (e.key === "Escape") closeModal();
      });

      modalOk.onclick = async () => {
        const value = (modalInput.value || "").trim();
        if (!value) return showModalError("Este campo no puede estar vacío.");
        if (typeof modalOnConfirm === "function") await modalOnConfirm(value);
      };

      editProjectContextBtn.onclick = async () => {
        if (!selectedProjectId) {
          alert("Selecciona un proyecto.");
          return;
        }

        try {
          setStatus("Cargando proyecto…");
          const proj = await loadProject(selectedProjectId);

          openModal({
            title: `Contexto del proyecto: ${proj.name}`,
            label: "Instrucciones / memoria del proyecto",
            placeholder:
              "Ej: Este proyecto es para un clon de ChatGPT. Responde siempre en español. Prioriza FastAPI…",
            okText: "Guardar",
            defaultValue: proj.context || "",
            onConfirm: async (contextText) => {
              try {
                setStatus("Guardando contexto…");
                await apiPatch(`/api/projects/${selectedProjectId}`, {
                  context: contextText,
                });
                setStatus("Listo");
                closeModal();
              } catch (e) {
                setStatus("Error");
                showModalError(e.message || "No se pudo guardar.");
              }
            },
          });

          setStatus("Listo");
        } catch (e) {
          setStatus("Error");
          alert(e.message);
        }
      };

      // -------------------------
      // API
      // -------------------------
      async function apiGet(url) {
        const res = await fetch(url);
        const data = await res.json();
        if (!res.ok) throw new Error(data.error || "Error");
        return data;
      }
      async function apiPost(url, body) {
        const res = await fetch(url, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(body),
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error || "Error");
        return data;
      }
      async function apiPatch(url, body) {
        const res = await fetch(url, {
          method: "PATCH",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(body),
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error || "Error");
        return data;
      }

      // -------------------------
      // Render
      // -------------------------
      function renderProjects() {
        projectsList.innerHTML = "";
        projectCount.textContent = projects.length;

        projects.forEach((p) => {
          const row = document.createElement("div");
          row.className =
            "sb-row" + (p.id === selectedProjectId ? " active" : "");
          row.innerHTML = `<div class="sb-row-title" title="${escapeHtml(
            p.name
          )}">${escapeHtml(p.name)}</div>`;

          row.onclick = async () => {
            selectedProjectId = p.id;
            selectedChatId = null;
            topTitle.textContent = "Selecciona un chat";
            renderProjects();
            await loadChats();
            clearChatView();
          };
          projectsList.appendChild(row);
        });
      }

      function renderChats(list = chats) {
        chatsList.innerHTML = "";
        chatCount.textContent = chats.length;

        list.forEach((c) => {
          const row = document.createElement("div");
          row.className = "sb-row" + (c.id === selectedChatId ? " active" : "");
          row.innerHTML = `<div class="sb-row-title" title="${escapeHtml(
            c.title
          )}">${escapeHtml(c.title)}</div>`;
          row.onclick = async () => {
            selectedChatId = c.id;
            topTitle.textContent = c.title;
            renderChats();
            await loadMessages();
          };
          chatsList.appendChild(row);
        });
      }

      function clearChatView() {
        chatInner.innerHTML = `
        <div class="msg assistant">
          <div class="bubble muted">Selecciona un chat para ver el historial.</div>
        </div>
      `;
      }

      function addMessage(role, text) {
        const row = document.createElement("div");
        row.className = `msg ${role === "user" ? "user" : "assistant"}`;
        row.innerHTML = `<div class="bubble">${escapeHtml(text)}</div>`;
        chatInner.appendChild(row);
        chatScroll.scrollTop = chatScroll.scrollHeight;
      }

      // -------------------------
      // Data loading
      // -------------------------
      async function loadProjects() {
        const data = await apiGet("/api/projects");
        projects = data.projects || [];
        if (!selectedProjectId && projects.length)
          selectedProjectId = projects[0].id;
        renderProjects();
      }

      async function loadChats() {
        if (!selectedProjectId) {
          chats = [];
          renderChats();
          return;
        }
        const data = await apiGet(`/api/projects/${selectedProjectId}/chats`);
        chats = data.chats || [];
        renderChats();

        if (!selectedChatId && chats.length) {
          selectedChatId = chats[0].id;
          topTitle.textContent = chats[0].title;
          renderChats();
          await loadMessages();
        }
      }

      async function loadMessages() {
        if (!selectedChatId) return clearChatView();

        const data = await apiGet(`/api/chats/${selectedChatId}/messages`);
        const msgs = data.messages || [];
        topTitle.textContent = data.chat?.title || "Chat";

        chatInner.innerHTML = "";
        if (msgs.length === 0) {
          chatInner.innerHTML = `
          <div class="msg assistant">
            <div class="bubble muted">Este chat está vacío. Escribe un mensaje abajo.</div>
          </div>
        `;
        } else {
          msgs.forEach((m) => addMessage(m.role, m.content));
        }
        chatScroll.scrollTop = chatScroll.scrollHeight;
      }

      // -------------------------
      // Actions
      // -------------------------
      newProjectBtn.onclick = () => {
        openModal({
          title: "Nuevo proyecto",
          label: "Nombre del proyecto",
          placeholder: "Ej: piloto",
          okText: "Crear",
          defaultValue: "",
          onConfirm: async (name) => {
            try {
              setStatus("Creando proyecto…");
              await apiPost("/api/projects", { name });
              await loadProjects();
              selectedProjectId = projects[0]?.id || selectedProjectId;
              selectedChatId = null;
              await loadChats();
              clearChatView();
              setStatus("Listo");
              closeModal();
            } catch (e) {
              setStatus("Error");
              showModalError(e.message || "No se pudo crear el proyecto.");
            }
          },
        });
      };

      newChatBtn.onclick = async () => {
        if (!selectedProjectId) return;

        try {
          setStatus("Creando chat…");
          const data = await apiPost(
            `/api/projects/${selectedProjectId}/chats`,
            { title: "New chat" }
          );
          await loadChats();
          selectedChatId = data.chat.id;
          await loadMessages();
          setStatus("Listo");
        } catch (e) {
          setStatus("Error");
          alert(e.message);
        }
      };

      renameChatBtn.onclick = () => {
        if (!selectedChatId) return;

        const current = topTitle.textContent || "Chat";
        openModal({
          title: "Renombrar chat",
          label: "Nuevo nombre",
          placeholder: "Nombre del chat",
          okText: "Guardar",
          defaultValue: current,
          onConfirm: async (title) => {
            try {
              setStatus("Renombrando…");
              await apiPatch(`/api/chats/${selectedChatId}`, { title });
              await loadChats();
              await loadMessages();
              setStatus("Listo");
              closeModal();
            } catch (e) {
              setStatus("Error");
              showModalError(e.message || "No se pudo renombrar.");
            }
          },
        });
      };

      async function sendMessage() {
        const text = input.value.trim();
        if (!text || !selectedChatId) return;

        addMessage("user", text);
        input.value = "";
        autoResize();
        setStatus("Pensando…");

        const typing = document.createElement("div");
        typing.className = "msg assistant";
        typing.innerHTML = `<div class="bubble muted">...</div>`;
        chatInner.appendChild(typing);
        chatScroll.scrollTop = chatScroll.scrollHeight;

        try {
          const data = await apiPost("/api/send", {
            chat_id: selectedChatId,
            message: text,
          });
          typing.remove();
          addMessage("assistant", data.reply);
          setStatus("Listo");
          await loadChats(); // por si auto-título cambió
          const currentChat = chats.find((c) => c.id === selectedChatId);
          if (currentChat) topTitle.textContent = currentChat.title;
        } catch (e) {
          typing.remove();
          addMessage("assistant", "Error: " + e.message);
          setStatus("Error");
        }
      }

      sendBtn.onclick = sendMessage;
      input.addEventListener("input", autoResize);
      input.addEventListener("keydown", (e) => {
        if (e.key === "Enter" && !e.shiftKey) {
          e.preventDefault();
          sendMessage();
        }
      });

      // Buscar chats (filtra en el frontend)
      searchChats.addEventListener("input", () => {
        const q = (searchChats.value || "").toLowerCase().trim();
        if (!q) return renderChats();
        const filtered = chats.filter((c) =>
          (c.title || "").toLowerCase().includes(q)
        );
        renderChats(filtered);
      });

      // Init
      (async function init() {
        try {
          setStatus("Cargando…");
          autoResize();
          await loadProjects();
          await loadChats();
          setStatus("Listo");
          input.focus();
        } catch (e) {
          setStatus("Error");
          alert(e.message);
        }
      })();
    </script>
